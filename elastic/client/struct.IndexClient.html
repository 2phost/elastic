Examples

Create an index called myindex:

let my_index = index("myindex");

let response = client.index(my_index).create().send()?;

assert!(response.acknowledged());

Create an index with settings and document mappings for a DocumentType called MyType:

let body = json!({
    "settings": {
        "index": {
            "number_of_shards": 3,
            "number_of_replicas": 2
        }
    },
    "mappings": {
        MyType::static_index(): MyType::index_mapping()
    }
});

let response = client.index("myindex")
                     .create()
                     .body(body.to_string())
                     .send()?;

assert!(response.acknowledged());

For more details on document types and mapping, see the types module.
